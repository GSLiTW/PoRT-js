  4 Decimal.set({ modulo: Decimal.ROUND_FLOOR });

  9 function divmod(a, b, n) {
 10   let aCopy = (Decimal.isDecimal(a)) ? a : Decimal(a);
 11   let bCopy = (Decimal.isDecimal(b)) ? b : Decimal(b);
 12   let nCopy = (Decimal.isDecimal(n)) ? n : Decimal(n);
 13   let t = Decimal('0');
 14   let nt = Decimal('1');
 15   let r = nCopy;
 16   let nr = bCopy.mod(n);
 17   let tmp;
 18   while (!nr.isZero()) {
 19     let quot = Decimal.floor(r.div(nr));
 20     tmp = nt;  nt = t.sub(quot.times(nt));  t = tmp;
 21     tmp = nr;  nr = r.sub(quot.times(nr));  r = tmp;
 22   };
 23   if (r.greaterThan(1)) return Decimal(0);
 24   if (t.isNegative()) t = t.add(n);
 25   return aCopy.times(t).mod(n);
 26 }

 28 function random(lower, upper) {
 29   if (lower > upper) {
 30     const temp = lower;
 31     lower = upper;
 32     upper = temp;
 33   }
 34 
 35   return lower.add(Decimal.random().times(upper.sub(lower + 1)).floor());
 36 }

 38 // Polynomial function where `a` is the coefficients
 39 function q(x, { a }) {
 40   let value = a[0];
 41   for (let i = 1; i < a.length; i++) {
 42     value = value.add(x.pow(i).times(a[i]));
 43   }
 44 
 45   return value;
 46 }

 48 function split(secret, n, k, prime) {
 49   if (Number.isInteger(secret) || Number.isInteger(prime)) {
 50     throw new TypeError(
 51       'The shamir.split() function must be called with a String<secret>' +
 52       'but got Number<secret>.'
 53     );
 54   }
 55 
 56   if (Number.isInteger(prime)) {
 57     throw new TypeError(
 58       'The shamir.split() function must be called with a String<prime>' +
 59       'but got Number<prime>.'
 60     );
 61   }
 62 
 63   if (secret.substring(0, 2) !== '0x') {
 64     throw new TypeError(
 65       'The shamir.split() function must be called with a' +
 66       'String<secret> in hexadecimals that begins with 0x.'
 67     );
 68   }
 69 
 70   if (!Decimal.isDecimal(prime) && prime.substring(0, 2) !== '0x') {
 71     throw new TypeError(
 72       'The shamir.split() function must be called with a' +
 73       'String<prime> in hexadecimals that begins with 0x.'
 74     );
 75   }
 76 
 77   const S = Decimal(secret);
 78   const p = Decimal(prime);
 79 
 80   if (S.greaterThan(prime)) {
 81     throw new RangeError('The String<secret> must be less than the String<prime>.');
 82   }
 83 
 84   let a = [S];
 85   let D = [];
 86 
 87   for (let i = 1; i < k; i++) {
 88     let coeff = random(Decimal(0), p.sub(0x1));
 89     a.push(coeff);
 90   }
 91 
 92   for (let i = 0; i < n; i++) {
 93     let x = Decimal(i + 1);
 94     D.push({
 95       x,
 96       y: q(x, { a }).mod(p),
 97     });
 98   }
 99 
100   return D.map((share) => ({
101     x: share.x.toString(),
102     y: share.y.toHex(),
103   }));
104 }

106 function lagrangeBasis(data, j) {
107   // Lagrange basis evaluated at 0, i.e. L(0).
108   // You don't need to interpolate the whole polynomial to get the secret, you
109   // only need the constant term.
110   let denominator = Decimal(1);
111   let numerator = Decimal(1);
112   for (let i = 0; i < data.length; i++) {
113     if (!data[j].x.equals(data[i].x)) {
114       denominator = denominator.times(data[i].x.minus(data[j].x));
115     }
116   }
117 
118   for (let i = 0; i < data.length; i++) {
119     if (!data[j].x.equals(data[i].x)) {
120       numerator = numerator.times(data[i].x);
121     }
122   }
123 
124   return {
125     numerator,
126     denominator,
127   };
128 }
129 
130 function lagrangeInterpolate(data, p) {
131   let S = Decimal(0);
132 
133   for (let i = 0; i < data.length; i++) {
134     let basis = lagrangeBasis(data, i);
135     S = S.add(data[i].y.times(divmod(basis.numerator, basis.denominator, p)));
136   }
137 
138   const rest = S.mod(p);
139 
140   return rest;
141 }

143 function combine(shares, prime) {
144   const p = Decimal(prime);
145 
146   // Wrap with Decimal on the input shares
147   const decimalShares = shares.map((share) => ({
148     x: Decimal(share.x),
149     y: Decimal(share.y),
150   }));
151 
152   return lagrangeInterpolate(decimalShares, p);
153 }
