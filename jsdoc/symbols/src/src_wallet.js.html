1 const sha256 = require("sha256");
  2 const secureRandom = require('secure-random');
  3 const elliptic = require('elliptic');
  4 const ripemd160 = require('ripemd160');
  5 const base58 = require('bs58');
  6 const BigInteger = require("bigi");
  7 const ec = new elliptic.ec('secp256k1');
  8 
  9 const CHECKSUM_LENGTH = 4; // 4 bytes
 10 /**
 11  * Generate & Initialize Wallet Class
 12  * @class
 13  * @constructor
 14  * @param  {string} [prik] - private key
 15  * @param  {string} [pubk] - public key
 16  */
 17 function Wallet(prik='', pubk='') {
 18     this.privateKey = prik;
 19     this.publicKey = pubk;
 20 
 21     if(prik == '' || pubk == ''){
 22         this.NewKeyPair();
 23     }
 24 
 25     this.signerPrivateData = {
 26         privateKey:BigInteger.fromHex(this.privateKey),
 27         session: null
 28     };
 29     this.publicKeyCompressed = ec.keyFromPublic(this.publicKey, "hex").getPublic().encodeCompressed("hex")
 30 };
 31 /**
 32  * Generate key pair when no parameters passed into wallet constructor
 33  */
 34 Wallet.prototype.NewKeyPair = function(){
 35     this.privateKey = secureRandom.randomBuffer(32);
 36     this.signerPrivateData.privateKey = BigInteger.fromBuffer(this.privateKey);
 37     const ecdsa = new elliptic.ec('secp256k1');
 38     const keys = ecdsa.keyFromPrivate(this.privateKey);
 39     this.publicKey = keys.getPublic('hex');     //integer
 40 }
 41 /**
 42  * Generate the hash of public key
 43  * @return {string}  public key hash
 44  */
 45 Wallet.prototype.PublicKeyHash = function(){
 46     let hash = sha256(Buffer.from(publicKey, 'hex'));
 47 
 48     let publicKeyHash = new ripemd160().update(Buffer.from(hash, 'hex')).digest();
 49 
 50     return publicKeyHash.toString('hex');
 51 }
 52 /**
 53  * Hash twice to generate checksum
 54  * @param  {string} versionedHash - bitcoin blockchain version byte + publickey hash
 55  * @return {string} - checksum for address conversion
 56  */
 57 Wallet.prototype.Checksum = function(versionedHash){
 58     var firstHash = sha256(Buffer.from(versionedHash, 'hex'));
 59     var secondHash = sha256(Buffer.from(firstHash, 'hex'));
 60     secondHash = Buffer.from(secondHash, 'hex');
 61     return secondHash.subarray(0,CHECKSUM_LENGTH);
 62 
 63 }
 64 /**
 65  * Generate address from initial public key
 66  * @return {string}  base58 address
 67  */
 68 Wallet.prototype.Address = function() {
 69     var pubHash = this.PublicKeyHash();
 70     var versionedHash = "00" + pubHash;
 71     var checksum = this.Checksum(versionedHash);
 72     var fullHash = versionedHash + checksum.toString('hex');
 73     
 74     return base58.encode(Buffer.from(fullHash,'hex'));
 75 }
 76 /**
 77  * Sign data for multisignature
 78  * @param  {string} dataHash - data to sign
 79  * @return {string} signature
 80  */
 81 Wallet.prototype.Sign = function(dataHash) {
 82     const ecdsa = new elliptic.ec('secp256k1');
 83     let signature = ecdsa.sign(dataHash, privateKey, "hex", {canonical:true});
 84     return signature;
 85 }
 86 /**
 87  * Validate if the input address is valid
 88  * @param  {string} address - address to validate
 89  * @return {bool} True if the input address is valid; False otherwise
 90  */
 91 Wallet.prototype.ValidateAddress = function(address){
 92     var pubHash = base58.decode(address);
 93     var actualChecksum = pubHash.subarray(pubHash.length - CHECKSUM_LENGTH);
 94     var version = pubHash.subarray(0,1);
 95     var pubHash = pubHash.subarray(1,pubHash.length - CHECKSUM_LENGTH);
 96     var targetChecksum = this.Checksum(Buffer.concat([version, pubHash]));
 97 
 98     return (actualChecksum.compare(targetChecksum) == 0)
 99 
100 }
101 
102 module.exports = Wallet;
