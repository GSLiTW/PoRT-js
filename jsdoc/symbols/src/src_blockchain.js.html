 52 Blockchain.prototype.chainIsValid = function(blockchain){
 53     var validChain = true;
 54     for(var i=1; i<blockchain.length; i++){
 55         const currentBlock = blockchain[i];
 56         const prevBlock = blockchain[i-1];
 57         const blockHash = this.hashBlock(prevBlock["hash"], {
 58             transactions: currentBlock["transactions"],
 59             index: currentBlock["height"]
 60             },
 61             currentBlock["nonce"]
 62         );
 63         if(blockHash.substring(0, 2) !== "00")
 64             validChain = false;
 65         if(currentBlock["previousBlockHash"] !== prevBlock["hash"])
 66             validChain = false;
 67     }
 68 
 69     const genesisBlock = blockchain[0];
 70     const correctNonce = genesisBlock["nonce"] === 100;
 71     const correctPreviousBlockHash = genesisBlock["previousBlockHash"] === "0";
 72     const correctHash = genesisBlock["hash"] === "0";
 73     const correctTransactions = genesisBlock["transactions"].length === 0;
 74 
 75     if(!correctNonce || !correctPreviousBlockHash || !correctHash || !correctTransactions)
 76         validChain = false;
 77     
 78     return validChain;
 79 }

114 Blockchain.prototype.getAddressData = function(address){
115     const addressTransactions = [];
116     this.chain.forEach(block =>{
117         block.transactions.forEach(transaction =>{
118             if(transaction.sender === address || transaction.recipient === address)
119             addressTransactions.push(transaction);
120         });
121     });
122 
123     let balance = 0;
124     addressTransactions.forEach(transaction =>{
125         if(transaction.recipient === address)
126             balance += transaction.amount;
127         if(transaction.sender === address)
128             balance -= transaction.amount;
129     });
130 
131     return {
132         addressTransactions: addressTransactions,
133         addressBalance : balance
134     }

151 Blockchain.prototype.executeTransactions = function(block) {
152     block.data.forEach(transaction => {
153         switch (transaction.type) {
154             case TRANSACTION_TYPE.transaction:
155                 this.accounts.update(transaction);
156                 this.accounts.transferFee(block, transaction);
157             break;
158             case TRANSACTION_TYPE.stake:
159             this.stakes.update(transaction);
160             this.accounts.decrement(
161                 transaction.input.from,
162                 transaction.output.amount
163             );
164             this.accounts.transferFee(block, transaction);
165             break;
166             case TRANSACTION_TYPE.validator_fee:
167             if (this.validators.update(transaction)) {
168                 this.accounts.decrement(
169                 transaction.input.from,
170                 transaction.output.amount
171                 );
172                 this.accounts.transferFee(block, transaction);
173             }
174             break;
175         }
176     });
177 }
