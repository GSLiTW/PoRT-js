  2 const randomBuffer = (len) => Buffer.from(randomBytes.sync(len));

 10 /**
 11  * Constructor of the Voter class
 12  * @class Voter is used to generate cosignature for block creation
 13  * @constructor
 14  * @param  {string} port - Network port number of the voter
 15  * @param  {string} pubKey - Wallet public key of the voter
 16  * @param  {MPT} MPT - Local Merkle Patricia Trie copy of the voter
 17  */
 18 function Voter(port, pubKey, MPT){
 19     this.MPT = MPT;
 20     this.port = port;
 21     this.pubKey = pubKey;
 22     this.pubKeyCompressed = ec.keyFromPublic(this.pubKey, "hex").getPublic().encodeCompressed("hex")
 23 }
 24 /**
 25  * Check if the caller is selected as voter to perform actions for the current round of block construction
 26  * @return {bool} True if the caller is the voter of the current round of block construction; False otherwise 
 27  */
 28 Voter.prototype.IsValid = function() {
 29     return (this.MPT.Verify(this.pubKey) == 2); // Check by validating the dirty bit in the latest account MPT
 30 }

 55 /**
 56  * Get the publicly known data for multisig from creator and store them in Voter's data structure
 57  * (ref: https://bitcoindev.network/pure-javascript-implementation-of-the-schnorr-bip/)
 58  * @param  {Array[string]} pubKeys - wallet public keys of other voters
 59  * @param  {*} message - shared message between creator/voter
 60  */
 61 Voter.prototype.GetPublicData = function(pubKeys, message) {
 62     this.pubKeys = pubKeys.slice();
 63     for(var i in pubKeys) {
 64         pubKeys[i] = Buffer.from(pubKeys[i], 'hex');
 65     }
 66     message = convert.hash(Buffer.from(JSON.stringify(message), 'utf8'));
 67     this.publicData = {
 68         pubKeys: pubKeys,
 69         message: message,
 70         pubKeyHash: null,
 71         pubKeyCombined: null,
 72         commitments: [],
 73         nonces: [],
 74         nonceCombined: null,
 75         partialSignatures: [],
 76         signature: null,
 77     }
 78     // console.log(this.publicData);
 79 
 80     this.publicData.pubKeyHash = muSig.computeEll(this.publicData.pubKeys);
 81     this.publicData.pubKeyCombined = muSig.pubKeyCombine(this.publicData.pubKeys, this.publicData.pubKeyHash);
 82 }
 83 /**
 84  * Create the private signing session for multisig <br>
 85  * (ref: https://bitcoindev.network/pure-javascript-implementation-of-the-schnorr-bip/)
 86  * @param  signerPrivateData
 87  * @return SignerSession if it is the first voter (idx == 0); null otherwise
 88  */
 89 Voter.prototype.PrivateSign = function(signerPrivateData) {
 90     this.signerPrivateData = signerPrivateData;
 91     const idx = this.pubKeys.indexOf(this.pubKeyCompressed);
 92     const sessionId = randomBuffer(32); // must never be reused between sessions!
 93     this.signerPrivateData.session = muSig.sessionInitialize(
 94         sessionId,
 95         this.signerPrivateData.privateKey,
 96         this.publicData.message,
 97         this.publicData.pubKeyCombined,
 98         this.publicData.pubKeyHash,
 99         idx
100     );
101 
102 
103     if(idx == 0) {
104         this.SignerSession = JSON.parse(JSON.stringify(this.signerPrivateData.session));
105         this.SignerSession.secretKey = null;
106         this.SignerSession.secretNonce = null;
107         return this.SignerSession;
108     } else {
109         return null;
110     }
111 }

113 /**
114  * Get SignerSession from the first voter and store it in each voter's SignerSession <br>
115  * (ref: https://bitcoindev.network/pure-javascript-implementation-of-the-schnorr-bip/)
116  * @param  {SignerSession} SignerSession - first voter's signer session
117  */
118 Voter.prototype.GetSignerSession = function(SignerSession) {
119     SignerSession.sessionId = Buffer.from(SignerSession.sessionId)
120     SignerSession.ell = Buffer.from(SignerSession.ell)
121     SignerSession.nonce = Buffer.from(SignerSession.nonce)
122     SignerSession.commitment = Buffer.from(SignerSession.commitment)
123     SignerSession.pubKeyCombined = Buffer.from(SignerSession.pubKeyCombined)
124     SignerSession.message = Buffer.from(SignerSession.message)
125     this.SignerSession = SignerSession;
126 }

128 /**
129  * Exchange each voter's commitment (communication round 1) through creator <br>
130  * (ref: https://bitcoindev.network/pure-javascript-implementation-of-the-schnorr-bip/)
131  * @param  {} commitments
132  */
133 Voter.prototype.ExchangeCommitment = function(commitments) {
134     for(var i in commitments) {
135         commitments[i] = Buffer.from(commitments[i])
136     }
137     this.publicData.commitments = commitments;
138 }
139 /**
140  * Get nonces from each voter (communication round 2) through creator <br>
141  * (ref: https://bitcoindev.network/pure-javascript-implementation-of-the-schnorr-bip/)
142  * @param  {Array} nonces
143  */
144 Voter.prototype.ExchangeNonce = function(nonces) {
145     for(var i in nonces) {
146         nonces[i] = Buffer.from(nonces[i])
147     }
148     this.publicData.nonces = nonces;
149     
150     this.publicData.nonceCombined = muSig.sessionNonceCombine(this.SignerSession, this.publicData.nonces);
151     this.signerPrivateData.session.nonceIsNegated = this.SignerSession.nonceIsNegated;
152 
153     this.PartialSign();
154 }
155 /**
156  * Generate partial signatures over the given message <br>
157  * (ref: https://bitcoindev.network/pure-javascript-implementation-of-the-schnorr-bip/)
158  */
159 Voter.prototype.PartialSign = function() {
160     this.signerPrivateData.session.partialSignature = BigInteger.fromHex(muSig.partialSign(this.signerPrivateData.session, this.publicData.message, this.publicData.nonceCombined, this.publicData.pubKeyCombined).toHex());
161 }
162 /**
163  * Exchange voter's partial signatures (communication round 3) through creator <br>
164  * (ref: https://bitcoindev.network/pure-javascript-implementation-of-the-schnorr-bip/)
165  * @param  {Array} partialsigns
166  */
167 Voter.prototype.ExchangePartialSign = function(partialsigns) { 
168     for(var i in partialsigns) {
169         partialsigns[i] = BigInteger.fromHex(partialsigns[i])
170     }
171     this.publicData.partialSignatures = partialsigns;
172 
173 
174     for (let i = 0; i < this.publicData.pubKeys.length; i++) {
175 
176         muSig.partialSigVerify(
177           this.SignerSession,
178           this.publicData.partialSignatures[i],
179           this.publicData.nonceCombined,
180           i,
181           this.publicData.pubKeys[i],
182           this.publicData.nonces[i]
183         );
184     }
185 
186     this.CombinePartialSign();
187 }
188 /**
189  * Combine partial signatures to full signature (s, R) that can be verified against combined public key P
190  * ,then verify as Schnorr signature (s, R) over the message m and public key P. <br>
191  * (ref: https://bitcoindev.network/pure-javascript-implementation-of-the-schnorr-bip/)
192  */
193 Voter.prototype.CombinePartialSign = function() {
194     this.publicData.signature = muSig.partialSigCombine(this.publicData.nonceCombined, this.publicData.partialSignatures);
195     this.VerifyCoSig();
196 }
197 /**
198  * Verify the signature as a normal Schnorr Signature (s, R) over message m and public key P <br>
199  * (ref: https://bitcoindev.network/pure-javascript-implementation-of-the-schnorr-bip/)
200  */
201 Voter.prototype.VerifyCoSig = function() {
202     console.log("CoSig:", this.publicData.signature.toString('hex'))
203     schnorr.verify(this.publicData.pubKeyCombined, this.publicData.message, this.publicData.signature);
204     console.log("Voter", this.port, "- Verified :)")
205 }
