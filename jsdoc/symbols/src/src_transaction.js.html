1 const TRANSACTION_FEE = 1;
  2 // const uuid = require('uuid/v1')
  3 const sha256 = require('sha256')
  4 const ec = require('elliptic')
  5 
  6 function Transaction(senderWallet, to, amount, type) {
  7     // this.id = uuid(),
  8     this.type = type,
  9 
 10     this.output = {
 11         to: to,
 12         amount: amount - TRANSACTION_FEE,
 13         fee: TRANSACTION_FEE
 14     }
 15 
 16     this.input = {
 17         timestamp: Date.now(),
 18         from: senderWallet.publicKey,
 19         signature: senderWallet.sign(sha256(JSON.stringify(this.output).toString()))
 20     }
 21 
 22 }
 23 
 24 Transaction.prototype.newTransaction = function(senderWallet, to, amount, type) {
 25     if (amount + TRANSACTION_FEE > senderWallet.balance) {
 26       console.log(`Amount : ${amount} exceeds the balance`);
 27       return;
 28     }
 29 
 30     return Transaction(senderWallet, to, amount, type);
 31 }
 32 
 33 
 34 
 35 Transaction.prototype.verifyTransaction = function() {
 36     return ec.ec.KeyFromPublic(this.input.from).verify(sha256(JSON.stringify(this.output).toString(), this.input.signature))
 37 }
 38 
 39 module.exports = Transaction;
