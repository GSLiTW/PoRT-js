1 const Block = require("./block.js");
  2 const PoRT = require("./PoRT.js");
  3 /**
  4  * Generate & Initialize Creator Class
  5  * @class Creater is responsible for creating blocks and communicate with voter to generate cosignature
  6  * @param  {string} port - Network port number of the creator
  7  * @param  {string} pubKey - Wallet public key of the creator
  8  * @param  {MPT} MPT - Local Merkle Patricia Trie copy of the creator
  9  */
 10 function Creator(port, pubKey, MPT) {
 11     this.MPT = MPT;
 12     this.port = port;
 13     this.pubKey = pubKey;
 14 }
 15 /**
 16  * Check if the caller is selected as creator to perform actions for the current round of block construction, by passing publickey into MPT function
 17  * @return {bool} True if the caller is the creator of the current round of block construction; False otherwise
 18  */
 19 Creator.prototype.IsValid = function () {
 20     return (this.MPT.Verify(this.pubKey) == 1);
 21 }
 22 /**
 23  * Create a Block, adding transactions, MPT and metadata into it
 24  * @param  {list} pendingTxs
 25  * @param  {Number} height
 26  * @param  {string} previousHash
 27  * @return {Block} created block
 28  */
 29 Creator.prototype.Create = function (pendingTxs, height, previousHash) {
 30     for (var i = 0; i < pendingTxs.length; i++) {
 31         this.MPT.UpdateValue(pendingTxs[i].sender, pendingTxs[i].receiver, pendingTxs[i].value);
 32     }
 33 
 34     this.block = new Block(height, pendingTxs.transactions, previousHash, this.MPT);
 35 
 36     return this.block;
 37 }
 38 /**
 39  * Get voter from network and add to the list
 40  * @param  {string} VoterUrl - Voter's network url
 41  * @param  {string} VoterPubKey - Wallet public key of voter
 42  */
 43 Creator.prototype.GetVoter = function (VoterUrl, VoterPubKey) {
 44     if (this.VoterUrl == null) {
 45         this.VoterUrl = [VoterUrl];
 46         this.VoterPubKey = [VoterPubKey];
 47     } else {
 48         this.VoterUrl.push(VoterUrl);
 49         this.VoterPubKey.push(VoterPubKey);
 50     }
 51 }
 52 /**
 53  * Get signersession from network , generated voter
 54  * @param  {SignerSession} SignerSession - from network, generataed by voter
 55  */
 56 Creator.prototype.GetSignerSession = function (SignerSession) {
 57     this.SignerSession = SignerSession;
 58 }
 59 /**
 60  * Store and Collect data(commitment) from voter through network (communication round 1) and to be sent back to voter through network <br>
 61  * (ref: https://bitcoindev.network/pure-javascript-implementation-of-the-schnorr-bip/)
 62  * @param  {} VoterCommitment 
 63  * @param  {string} VoterPubKey - Corresponding Voter's public key, from network
 64  */
 65 Creator.prototype.GetCommitments = function (VoterCommitment, VoterPubKey) {
 66     var idx = this.VoterPubKey.indexOf(VoterPubKey);
 67     if (this.commitments == null) {
 68         this.commitments = []
 69         for (var i = 0; i < this.VoterPubKey.length; i++) {
 70             this.commitments.push(null);
 71         }
 72     }
 73 
 74     this.commitments[idx] = VoterCommitment;
 75     for (var i in this.commitments) {
 76         if (this.commitments[i] == null) {
 77             return false;
 78         }
 79     }
 80 
 81     return true;
 82 }
 83 /**
 84  * Store and Collect Nonce from voter through network (communication round 2) and to be sent back to voter through network <br>
 85  * (ref: https://bitcoindev.network/pure-javascript-implementation-of-the-schnorr-bip/)
 86  * @param  {Number} VoterNonce
 87  * @param  {string} VoterPubKey
 88  */
 89 Creator.prototype.GetNonces = function (VoterNonce, VoterPubKey) {
 90     var idx = this.VoterPubKey.indexOf(VoterPubKey);
 91     if (this.nonces == null) {
 92         this.nonces = []
 93         for (var i = 0; i < this.VoterPubKey.length; i++) {
 94             this.nonces.push(null);
 95         }
 96     }
 97 
 98     this.nonces[idx] = VoterNonce;
 99     for (var i in this.nonces) {
100         if (this.nonces[i] == null) {
101             return false;
102         }
103     }
104 
105     return true;
106 }
107 /**
108  * Store and Collect PartialSign from voters through network (communication round 3) and to be sent back to voter through network <br>
109  * (ref: https://bitcoindev.network/pure-javascript-implementation-of-the-schnorr-bip/)
110  * @param  {string} VoterPartialSign
111  * @param  {string} VoterPubKey
112  */
113 Creator.prototype.GetPartialSigns = function (VoterPartialSign, VoterPubKey) {
114     var idx = this.VoterPubKey.indexOf(VoterPubKey);
115     if (this.partialsigns == null) {
116         this.partialsigns = []
117         for (var i = 0; i < this.VoterPubKey.length; i++) {
118             this.partialsigns.push(null);
119         }
120     }
121 
122     this.partialsigns[idx] = VoterPartialSign;
123     for (var i in this.partialsigns) {
124         if (this.partialsigns[i] == null) {
125             return false;
126         }
127     }
128 
129     return true;
130 }
131 /**
132  * The final step of communication: store cosignature in the block
133  * @param  {string} cosig - cosignature generated from voter.combinepartialsign
134  */
135 Creator.prototype.GetCosig = function (cosig) {
136     this.block.coSignature = cosig; 
137 }
138 /**
139  * Complete the generation of current new block
140  * @param  {string} previousHash - hash value of the last block
141  * @return the completed new block
142  */
143 Creator.prototype.GetBlock = function (previousHash) {
144     var creatorPoRT = new PoRT(this.pubKey, this.MPT, 1);
145     this.block.nextCreator = creatorPoRT.next_maintainer[1];
146     for (var i = 0; i < this.VoterPubKey.length; i++) {
147         var voterPoRT = new PoRT(this.VoterPubKey[i], this.MPT, 2);
148         this.block.nextVoters.push(voterPoRT.next_maintainer[1]);
149     }
150     this.block.hash = this.block.hashBlock(previousHash, this.block);
151     return this.block;
152 }
