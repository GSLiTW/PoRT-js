1 const Block = require("./block.js");
  2 const PoRT = require("./PoRT.js");
  3 
  4 function Creator(port, pubKey, MPT) {
  5     this.MPT = MPT;
  6     this.port = port;
  7     this.pubKey = pubKey;
  8 }
  9 
 10 Creator.prototype.IsValid = function () {
 11     return (this.MPT.Verify(this.pubKey) == 1);
 12 }
 13 
 14 Creator.prototype.Create = function (pendingTxs, height, previousHash) {
 15     for (var i = 0; i < pendingTxs.length; i++) {
 16         this.MPT.UpdateValue(pendingTxs[i].sender, pendingTxs[i].receiver, pendingTxs[i].value);
 17     }
 18 
 19     this.block = new Block(height, pendingTxs.transactions, previousHash, this.MPT);
 20 
 21     return this.block;
 22 }
 23 
 24 Creator.prototype.GetVoter = function (VoterUrl, VoterPubKey) {
 25     if (this.VoterUrl == null) {
 26         this.VoterUrl = [VoterUrl];
 27         this.VoterPubKey = [VoterPubKey];
 28     } else {
 29         this.VoterUrl.push(VoterUrl);
 30         this.VoterPubKey.push(VoterPubKey);
 31     }
 32 }

 38 Creator.prototype.GetCommitments = function (VoterCommitment, VoterPubKey) {
 39     var idx = this.VoterPubKey.indexOf(VoterPubKey);
 40     if (this.commitments == null) {
 41         this.commitments = []
 42         for (var i = 0; i < this.VoterPubKey.length; i++) {
 43             this.commitments.push(null);
 44         }
 45     }
 46 
 47     this.commitments[idx] = VoterCommitment;
 48     for (var i in this.commitments) {
 49         if (this.commitments[i] == null) {
 50             return false;
 51         }
 52     }
 53 
 54     return true;
 55 }

 99 Creator.prototype.GetBlock = function (previousHash) {
100     var creatorPoRT = new PoRT(this.pubKey, this.MPT, 1);
101     this.block.nextCreator = creatorPoRT.next_maintainer[1];
102     for (var i = 0; i < this.VoterPubKey.length; i++) {
103         var voterPoRT = new PoRT(this.VoterPubKey[i], this.MPT, 2);
104         this.block.nextVoters.push(voterPoRT.next_maintainer[1]);
105     }
106     this.block.hash = this.block.hashBlock(previousHash, this.block);
107     return this.block;
108 }
