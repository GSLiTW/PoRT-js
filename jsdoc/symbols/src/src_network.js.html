334 app.post("/receive-new-block", function (req, res) {
335     const seq = req.body.SeqNum;
336     if (seqList.indexOf(seq) == -1) {
337         const newBlock = req.body.newBlock;
338         const lastBlock = chain.getLastBlock();
339         const correctHash = lastBlock.hash === newBlock.previousBlockHash;
340         const correctIndex = lastBlock["height"] + 1 == newBlock["height"];
341 
342 
343         for (var i = 0, UpdateList = newBlock.transactions; i < UpdateList.length; i++) {
344             Tree.UpdateValue(UpdateList[i].sender, UpdateList[i].receiver, parseFloat(UpdateList[i].value));
345         }
346 
347         Tree.UpdateDbit(lastBlock.nextCreator, 0);
348         Tree.UpdateDbit(newBlock.nextCreator, 1);
349 
350         for (var i = 0; i < newBlock.nextVoters.length; i++) {
351             Tree.UpdateDbit(lastBlock.nextVoters[i], 0);
352             Tree.UpdateDbit(newBlock.nextVoters[i], 2);
353         }
354 
355         if (correctHash && correctIndex) {
356             chain.chain.push(newBlock);
357             pending_txn_pool.clean();
358             if (newBlock.height == 4000719) pending_txn_pool.create(3);
359             var currentdate = new Date();
360             var datetime = "Last Sync: " + currentdate.getDate() + "/"
361                 + (currentdate.getMonth() + 1) + "/"
362                 + currentdate.getFullYear() + " @ "
363                 + currentdate.getHours() + ":"
364                 + currentdate.getMinutes() + ":"
365                 + currentdate.getSeconds() + "."
366                 + currentdate.getMilliseconds();
367             console.log(datetime);
368             res.json({
369                 note: 'New block received and accepted.',
370                 newBlock: newBlock
371             });
372 
373             // refund creator's & voter's tax
374             if (lastBlock["height"] >= 4000718) {
375                 Tree.RefundTax(lastBlock.nextCreator, Tree.Search(lastBlock.nextCreator)[1]);
376                 for (var i = 0; i < lastBlock.nextVoters.length; i++) {
377                     Tree.RefundTax(lastBlock.nextVoters[i], (Tree.Search(lastBlock.nextVoters[i])[1]) * 0.7);
378                 }
379             }
380         }
381         else {
382             res.json({
383                 note: 'New block rejected.',
384                 newBlock: newBlock
385             });
386         }
387 
388         seqList.push(seq);
389 
390 
391         const requestPromises = [];
392         chain.networkNodes.forEach(networkNodeUrl => {
393             const requestOptions = {
394                 uri: networkNodeUrl + "/receive-new-block",
395                 method: "POST",
396                 body: { SeqNum: seq, newBlock: newBlock },
