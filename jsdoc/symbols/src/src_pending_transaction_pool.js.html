1 // data structure for pending txns
  2 const CSV_data = require("./CSV_data.js");
  3 const Transaction_MT = require("./transaction_for_mapping_table.js");
  4 
  5 function Pending_Transaction_Pool(tx = []) {
  6     this.transactions = tx;
  7 };
  8 
  9 Pending_Transaction_Pool.prototype.create = function(num) {
 10     var data = new CSV_data();
 11     var data_ = data.getData(num); //get data of block1
 12     if(num == 1) {
 13         for(var i = 1; i < 44; i++) {
 14             var txn = new Transaction_MT(data_[i][0], data_[i][2], data_[i][3], data_[i][4]);
 15             this.transactions.push(txn);
 16         }
 17     }
 18     else if(num == 2) {
 19         for(var i = 1; i < 44; i++) {
 20             var txn = new Transaction_MT(data_[i][0], data_[i][2], data_[i][3], data_[i][4]);
 21             this.transactions.push(txn);
 22         }
 23     }
 24     else if(num == 3) {
 25         for(var i = 1; i < 50; i++) {
 26             var txn = new Transaction_MT(data_[i][0], data_[i][2], data_[i][3], data_[i][4]);
 27             this.transactions.push(txn);
 28         }
 29     }
 30     else console.log("wrong block number.");
 31 }
 32 
 33 Pending_Transaction_Pool.prototype.clean = function() {
 34     this.transactions = [];
 35 }
 36 
 37 Pending_Transaction_Pool.prototype.get_transaction = function() {
 38     return this.transactions;
 39 }
 40 
 41 Pending_Transaction_Pool.prototype.get_num_of_transaction = function() {
 42     return this.transactions.length;
 43 }
 44 
 45 module.exports = Pending_Transaction_Pool;
